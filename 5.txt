#include <iostream>

using namespace std;

int main() {
	setlocale(0, "");
	int number;
	cout << "Введите число: ";
	cin >> number;

	bool isPowerOfTwo = false;

	if (number > 0) {
		while (number != 1) {
			if (number % 2 != 0) {
				break;
			}
			number /= 2;
		}

		if (number == 1) {
			isPowerOfTwo = true;
		}
	}

	if (isPowerOfTwo) {
		cout << "Число " << number << " является степенью двойки" << endl;
	}
	else {
		cout << "Число " << number << " не является степенью двойки" << endl;
	}

	return 0;
}
